import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

public class SILab1Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();

        List<String> list = new ArrayList<>();

        for (int i = 0; i < n; i++) {
            list.add(sc.next());
        }
        
        // Calls the filterOddLengthStrings method from SILab1_1 and prints the result.
        System.out.println(SILab1_1.filterOddLengthStrings(list));
        
        // Calls the makeBigStringFrom method from SILab1_2 and prints the result.
        System.out.println(SILab1_2.makeBigStringFrom(list));
    } 
}
